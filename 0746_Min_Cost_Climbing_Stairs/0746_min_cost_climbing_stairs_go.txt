func min(num1, num2 int) int {
    if num1 < num2 {
        return num1
    }
    return num2
}

func minCostClimbingStairs(cost []int) int {
    prev := 0
    curr := 0
    n := len(cost)

    for i := 2; i<=n; i++ {
       next := min(cost[i - 2] + prev, cost[i - 1] + curr);
       prev = curr;
       curr = next;
    }

    return curr
}
