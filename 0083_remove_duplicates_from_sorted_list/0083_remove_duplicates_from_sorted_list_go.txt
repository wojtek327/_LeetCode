/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func deleteDuplicates(head *ListNode) *ListNode {
    if head == nil || head.Next == nil {
        return head
    }

    temp := head.Next
    lastValid := head

    for temp != nil {
        if temp.Val != lastValid.Val {
            lastValid.Next = temp
            lastValid = temp
        }
        temp = temp.Next
    }

    lastValid.Next = nil
    return head
}
