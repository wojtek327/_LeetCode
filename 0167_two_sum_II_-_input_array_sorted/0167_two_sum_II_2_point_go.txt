func twoSum(numbers []int, target int) []int {
    posRet := []int {-1, -1}

    left := 0;
    right := len(numbers) - 1;

    for left < right {
        sum := numbers[left] + numbers[right];

        if sum == target  {
            posRet[0] = left + 1;
            posRet[1] = right + 1;

            return posRet
        }

        if sum > target {
            right--;
        } else {
            left++;
        }
    }
    return posRet;
}
