func min(a, b int) int {
    if a < b { 
        return a
    }
    return b
}

func maxArea(height []int) int {
    l := 0
    r := len(height) - 1
    maxArea := 0

    for l<=r {
        tmpmaxArea := (r - l) * min(height[l], height[r]) 

        if tmpmaxArea > maxArea {
            maxArea = tmpmaxArea
        }

        if height[l]>height[r] {
            r--;
        } else {
            l++;
        }

    }

    return maxArea
}
