func majorityElement(nums []int) int {
    candidate := 0;
    counter := 0;
  
    for i := 0; i < len(nums); i++ {
        if counter == 0  {
            candidate = nums[i]
        }

        if candidate == nums[i]  { 
            counter += 1; 
        } else { 
            counter -= 1; 
        }
    }

    return candidate
}
